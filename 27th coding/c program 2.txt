#include <stdio.h>
#define MAXSIZE 10
 
void main()
{
    int array[MAXSIZE];
    int i, j, num, temp,c=0;
 
    printf("Enter the value of num \n");
    scanf("%d", &num);
    printf("Enter the array elements \n");
    for (i = 0; i < num; i++)
    {
        scanf("%d", &array[i]);
    }
    
    for (i = 0; i < num; i++)
    {
         for (j = 0; j < (num - i - 1); j++)
        {
            if (array[j] > array[j + 1])
            {
                temp = array[j];
                array[j] = array[j + 1];
                array[j + 1] = temp;
                c++;
            }
        }
    }
    printf("Sorted array is...\n");
    for (i = 0; i < num; i++)
    {
        printf("%d\n", array[i]);
    }
    printf("Num of iteration %d",c);
}
